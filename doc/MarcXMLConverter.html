            make(:date) do |date|
              date.label = 'creation'
              date.date_type = {'i' => 'inclusive',

              if control[7..10] && control[7..10].length == 4

              resource.dates << date
            end
          end

        # ID_0, ID_1, ID_2, ID_3
        "datafield[@tag='852']" => Proc.new {|resource, node|
          id = concatenate_subfields(%w(k h i m), node, '_')
          resource.id_0 = id unless id.empty?
        },


        "datafield[@tag='090']" => Proc.new {|resource, node|
          if resource.id_0.nil? or resource.id_0.empty?

def self.for_subjects_and_agents_only(input_file)
  new(input_file).instance_eval do
    @batch.record_filter = ->(record) {
      AgentManager.known_agent_type?(record.class.record_type) ||
      record.class.record_type == 'subject'
    }

    self
  end
end

def self.instance_for(type, input_file)
  if type == "marcxml"
    self.new(input_file)
  elsif type == "marcxml_subjects_and_agents"

  else
    nil
  end
end
