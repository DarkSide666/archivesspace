def initialize(opts = {})
  opts.each do |k,v|
    instance_variable_set("@#{k}", v)
  end

  @must_be_unique = ['subject']

  @record_filter = ->(record) { true }

  @uri_remapping = {}
  @seen_records = {}

  @working_file = opts[:working_file] || Tempfile.new("import_batch_working_file")
  @working_area = []
end

def flush_last
  last = @working_area.pop

  if last.class.method_defined? :uri and !last.uri.nil?
    _push(last)
  end
end
